	ORG $2ACDE

	BYTE 0xC0 0x46 0xC0 0x46 0xC0 0x46 0x00 0x28

	ORG $2AC28
	BYTE 0x49 0x10 0x0A 0x18 0x23 0x1C 0x66 0x33 0xFF 0x28 0x0B 0xD0
	
	ORG Free_Space
	#incbin "NoCrit.dmp"
	POIN Item_Table
	Hoplon:
	#incbin "HoplonModification.dmp"
	POIN Item_Table
